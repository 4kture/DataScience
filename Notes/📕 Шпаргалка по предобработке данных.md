    if pd.isna(text) or text.strip() == "":  
        return "—Ç–µ–∫—Å—Ç –æ—Ç—Å—É—Ç—Å—Ç–≤—É–µ—Ç"

    text = text.lower()  # –ü–µ—Ä–µ–≤–æ–¥ –≤ –Ω–∏–∂–Ω–∏–π —Ä–µ–≥–∏—Å—Ç—Ä

    text = re.sub(r"[^\w\s,.!?:;]", "", text)  # –£–¥–∞–ª–µ–Ω–∏–µ –ª–∏—à–Ω–∏—Ö —Å–∏–º–≤–æ–ª–æ–≤
    text = re.sub(r"\d+", "", text)  # –£–¥–∞–ª–µ–Ω–∏–µ —á–∏—Å–µ–ª
    text = re.sub(r"\s+", " ", text).strip()  # –£–¥–∞–ª–µ–Ω–∏–µ –ª–∏—à–Ω–∏—Ö –ø—Ä–æ–±–µ–ª–æ–≤

    stop_words = # –ü–æ–¥–∫–ª—é—á–µ–Ω–∏–µ —Ñ–∞–π–ª–∞ —Å–æ —Å—Ç–æ–ø-—Å–ª–æ–≤–∞–º–∏  
    words = text.split()
    words = [word for word in words if word not in stop_words]

    return " ".join(words)

df["X"] = df["X"].apply(preprocess_text)
